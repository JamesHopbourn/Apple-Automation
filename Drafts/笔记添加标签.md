
#### Script
```
var head = draft.processTemplate("[[line|1]]");
var content = draft.processTemplate("[[line|2..]]");

var text = draft.content;
if (text == ''){
  text = getClipboard()
  line = text.split("\n")
  head = line[0]
  content = text.replace(line[0],'')
}

// 替换中文引号
content = content.replace(/“/g,"「");
content = content.replace(/”/g,"」");

// 处理微信摘抄文章里面 HTML 的特殊空格
content = content.replace(/\u00a0/gm, "");
// 当有两个换行合并为一个
content = content.replace(/\n{2,}/g, '\n\n');
// 删除行首行末空格
head = head.replace(/(^\s*)|(\s*$)/g, "");
content = content.replace(/(^\s*)|(\s*$)/g, "");

draft.defineTag("head", head);
draft.defineTag("content", content);
```
